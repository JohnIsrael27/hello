2. Create requirements.txt
Flask==2.2.5

3. Create Dockerfile
FROM python:3.9-slim

WORKDIR /app

# Copy and install dependencies first (build cache friendly)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY . .

# Expose Flask port
EXPOSE 5000

# Run the application
CMD ["python", "app.py"]

4. Build the Docker image

Open a terminal in the project folder and run:

docker build -t flask-docker-app .

5. Run the container locally
If port 5000 is free:
docker run -p 5000:5000 flask-docker-app
# Visit: http://localhost:5000

If port 5000 is occupied and you want to map to host port 5050:
docker run -p 5050:5000 flask-docker-app
# Visit: http://localhost:5050

6. Login to Docker Hub
docker login

7. Tag the Docker image (replace with your Docker Hub username)
docker tag flask-docker-app your-dockerhub-username/flask-docker-app:latest

8. Push the image to Docker Hub
docker push your-dockerhub-username/flask-docker-app:latest

9. Test on any machine
# Pull the image
docker pull your-dockerhub-username/flask-docker-app:latest

# Run and map host port 5050 to container port 5000
docker run -p 5050:5000 your-dockerhub-username/flask-docker-app:latest
# Visit: http://<host-ip>:5050
